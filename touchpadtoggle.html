#!/bin/bash

TOUCHPAD_ACTIVE=`synclient -l | grep TouchpadOff | awk -F"=" '{print $2}'`
synclient TouchpadOff=$(synclient -l | grep -c 'TouchpadOff.*=.*0')
if [ $TOUCHPAD_ACTIVE = "1" ]; then
    notify-send "Touchpad Status" "Touchpad on" "--icon=$HOME/Pictures/icons/touchpadon.png"
else
    notify-send "Touchpad Status" "Touchpad off" "--icon=$HOME/Pictures/icons/touchpadoff.png"
fi

Then bind this script with a hotkey you like to use, such as XF86Touchpad, which represents Fn + F6, any other keys' name? just xbindkeys -k...
